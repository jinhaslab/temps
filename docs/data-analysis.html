## data import
mm1 = readRDS("data/mm1.rds")
############ function start ####################################
# IF you need to generate new variable, you can modify model_dat ahead of running smryf. 
model_dat = mm1 #%>%  mutate(w01a012 = case_when(w01a012 %in% c(2, 3) ~ 2, TRUE ~ w01a012))
smryf = function(codes,  ref_levels) {
# basic table 천명당 발생률 btab
values_sources = astt %>% filter(code == codes)
btab = model_dat %>%
  select(codes, fup, cvd_oc) %>%
  mutate(indep = get(codes)) %>%
  group_by(indep) %>%
  summarise(tpyears = sum(fup), 
            event   = sum(cvd_oc))  %>%
  mutate(incidence = event/tpyears*1000) %>%
  mutate(tpyears = round(tpyears), 
         event   = round(event), 
         incidence= sprintf("%.2f", incidence)) %>%
  mutate(values = as.character(indep))

mod1 = model_dat %>%
  select(w01gender1, agegp, codes, fup, cvd_oc) %>%
  mutate(indep = get(codes)) %>%
  coxph(data=., 
        Surv(fup, cvd_oc) ~
          relevel(as.factor(indep), ref = ref_levels) 
  )
gg1 = summary(mod1)$conf.int %>% data.frame() %>%
  mutate(values = rownames(.))  %>%
  mutate(values = str_replace_all(values, ".*ref_levels\\)", "")) %>%
  select(values, HR = exp.coef., LL =lower..95, UL =upper..95) %>%
  `rownames<-`( NULL ) %>%
  mutate(HR = sprintf("%.2f", ceiling_dec(HR, 2)), 
         LL = sprintf("%.2f", ceiling_dec(LL, 2)), 
         UL = sprintf("%.2f", ceiling_dec(UL, 2))) %>%
  mutate(Crude = sprintf("%s (%s-%s)", HR, LL, UL)) %>%
  select(values, Crude)
mod2 = model_dat %>%
  select(w01gender1, agegp, codes, fup, cvd_oc) %>%
  mutate(indep = get(codes)) %>%
  coxph(data=., 
        Surv(fup, cvd_oc) ~
          relevel(as.factor(indep), ref = ref_levels) +
          agegp + w01gender1
  )
gg2 = summary(mod2)$conf.int %>% data.frame() %>%
  mutate(values = rownames(.)) %>%
  mutate(values = str_replace_all(values, ".*ref_levels\\)", "")) %>%
  select(values, HR = exp.coef., LL =lower..95, UL =upper..95) %>%
  `rownames<-`( NULL ) %>%
  mutate(HR = sprintf("%.2f", ceiling_dec(HR, 2)), 
         LL = sprintf("%.2f", ceiling_dec(LL, 2)), 
         UL = sprintf("%.2f", ceiling_dec(UL, 2))) %>%
  mutate(`Model I` = sprintf("%s (%s-%s)", HR, LL, UL)) %>%
  select(values, `Model I`)
ref_dat = tibble("values" = ref_levels, "Crude" = "1.00 (reference)", "Model I" = "1.00 (reference)")
ftab = gg1  %>%
  right_join(gg2, by = "values") %>%
  filter(!values %in% c("agegp≥65", "w01gender1")) %>%
  rbind(ref_dat) %>%
  arrange(values) %>%
  mutate(code = codes) %>%
  left_join(as5, by=c("code")) %>%
  left_join(btab, by=c("values")) %>%
  left_join(values_sources, by=c("code", "values")) %>%
  select( source, code,values,value_sources,tpyears,event, incidence,  Crude, `Model I`) 
ss = ftab %>% pull(source) %>% unique()
ftab_html = ftab %>%
  select(`변수 내용`=value_sources,
    `관찰인년`=tpyears,
    `발생 건수`=event, `발생률` = incidence,  Crude, `Model I`) %>%
  kbl("html", caption = sprintf("%s과 뇌심혈관 질병 발생 및 사망과의 관계", ss)) %>%
  kable_classic_2(html_font = "Times") %>%
  footnote(number = c("발생률: 1000 인년 당 발생률", 
                      "Model I: 연령과 성별을 보정한 결과"))
return(list(ftab, ftab_html))
}
